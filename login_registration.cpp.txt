#include <iostream>
#include <fstream>
#include <string>
#include <sstream>

using namespace std;

// Simple hash function for password (not cryptographically secure)
string hashPassword(string password) {
    int hash = 0;
    for (size_t i = 0; i < password.length(); i++) {
        hash = hash + (int)password[i] * (int)(i + 1);
    }
    return to_string(hash);
}

// Check if username already exists
bool usernameExists(string username) {
    ifstream file("users.txt");
    
    if (!file.is_open()) {
        return false; // File doesn't exist, so username doesn't exist
    }
    
    string line;
    while (getline(file, line)) {
        stringstream ss(line);
        string fileUsername;
        getline(ss, fileUsername, ',');
        
        if (fileUsername == username) {
            file.close();
            return true;
        }
    }
    
    file.close();
    return false;
}

// Register a new user
void registerUser() {
    string username;
    string password;
    string confirmPassword;
    
    cout << "\n--- Registration ---\n";
    
    cout << "Enter username: ";
    cin >> username;
    
    // Validate username
    if (username.length() < 3) {
        cout << "Error: Username must be at least 3 characters.\n";
        return;
    }
    
    // Check if username already exists
    if (usernameExists(username)) {
        cout << "Error: Username already exists. Please choose a different username.\n";
        return;
    }
    
    cout << "Enter password: ";
    cin >> password;
    
    // Validate password
    if (password.length() < 4) {
        cout << "Error: Password must be at least 4 characters.\n";
        return;
    }
    
    cout << "Confirm password: ";
    cin >> confirmPassword;
    
    // Check if passwords match
    if (password != confirmPassword) {
        cout << "Error: Passwords do not match.\n";
        return;
    }
    
    // Hash password and save to file
    string hashedPassword = hashPassword(password);
    
    ofstream file("users.txt", ios::app);
    if (!file.is_open()) {
        cout << "Error: Could not open file.\n";
        return;
    }
    
    file << username << "," << hashedPassword << "\n";
    file.close();
    
    cout << "Success: User registered successfully!\n";
}

// Login existing user
void loginUser() {
    string username;
    string password;
    
    cout << "\n--- Login ---\n";
    
    cout << "Enter username: ";
    cin >> username;
    
    cout << "Enter password: ";
    cin >> password;
    
    // Hash the entered password
    string hashedPassword = hashPassword(password);
    
    // Read users file and verify
    ifstream file("users.txt");
    
    if (!file.is_open()) {
        cout << "Error: No users registered yet.\n";
        return;
    }
    
    string line;
    bool found = false;
    
    while (getline(file, line)) {
        stringstream ss(line);
        string fileUsername;
        string filePassword;
        
        getline(ss, fileUsername, ',');
        getline(ss, filePassword, ',');
        
        if (fileUsername == username && filePassword == hashedPassword) {
            found = true;
            break;
        }
    }
    
    file.close();
    
    if (found) {
        cout << "Success: Login successful! Welcome " << username << "!\n";
    } else {
        cout << "Error: Invalid username or password.\n";
    }
}

// Display menu and get user choice
void displayMenu() {
    cout << "\n=== User Authentication System ===\n";
    cout << "1. Register\n";
    cout << "2. Login\n";
    cout << "3. Exit\n";
    cout << "Enter your choice: ";
}

int main() {
    int choice;
    
    while (true) {
        displayMenu();
        
        if (!(cin >> choice)) {
            // Input failed - clear the error state and buffer
            cin.clear();
            cin.ignore(10000, '\n');
            cout << "Error: Invalid choice. Please try again.\n";
            continue;
        }
        
        if (choice == 1) {
            registerUser();
        } else if (choice == 2) {
            loginUser();
        } else if (choice == 3) {
            cout << "Thank you for using the system. Goodbye!\n";
            break;
        } else {
            cout << "Error: Invalid choice. Please try again.\n";
        }
    }
    
    return 0;
}
